% --- Custom Commands for Resume Elements ---
\newcommand{\rjobtitle}[1]{%
    \vspace{0.4em}\noindent\normalsize{\textbf{#1}}\par
}

\newcommand{\rcompanydate}[2]{%
    \noindent\normalsize{#1} \enspace $\cdot$ \enspace #2\par
    \vspace{0.2em}
}

\newcommand{\rbullet}[1]{%
    \item \small{#1}
}

\newcommand{\rinlinelists}[1]{%
    \small{#1}\par\vspace{0.5em}
}

\newcommand{\rhorizontalbullets}[1]{%
    \small{#1}\par\vspace{0.5em}
}

\newcommand{\rdynamiccolumns}[2]{%
    \small
    \begin{tabular}{#1}
        #2
    \end{tabular}
    \par\vspace{0.5em}
}

\newcommand{\reducationentry}[3]{%
    \vspace{0.4em}\noindent\normalsize{\textbf{#1}, \textit{#2} - #3}\par
}

% --- Header ---
\begin{center}
    {\fontsize{16pt}{19pt}\selectfont\bfseries \VAR{contact_info.name}} \\
    \vspace{0.3em}
    \normalsize{%
        \faMapMarker* \enspace \VAR{contact_info.location} \enspace $\cdot$ \enspace
        \faPhone* \enspace \VAR{contact_info.phone} \enspace $\cdot$ \enspace
        \faEnvelope \enspace \href{mailto:\VAR{contact_info.email}}{\VAR{contact_info.email}}%
        \BLOCK{if contact_info.social_links}%
            \BLOCK{for social_link in contact_info.social_links}%
                \BLOCK{if social_link.url and social_link.url.strip() and social_link.platform}%
                    \enspace $\cdot$ \enspace
                    \BLOCK{if social_link.platform == 'linkedin'}\faLinkedin%
                    \BLOCK{elif social_link.platform == 'github'}\faGithub%
                    \BLOCK{elif social_link.platform == 'twitter'}\faTwitter%
                    \BLOCK{elif social_link.platform == 'website'}\faGlobe%
                    \BLOCK{elif social_link.platform == 'stackoverflow'}\faStackOverflow%
                    \BLOCK{elif social_link.platform == 'medium'}\faMedium%
                    \BLOCK{elif social_link.platform == 'youtube'}\faYoutube%
                    \BLOCK{elif social_link.platform == 'behance'}\faBehance%
                    \BLOCK{elif social_link.platform == 'dribbble'}\faDribbble%
                    \BLOCK{elif social_link.platform == 'pinterest'}\faPinterest%
                    \BLOCK{else}\faLink%
                    \BLOCK{endif}%
                    \enspace \href{\VAR{social_link.url}}{\VAR{social_link.display_text or social_link.handle or social_link.platform}}%
                \BLOCK{endif}%
            \BLOCK{endfor}%
        \BLOCK{elif contact_info.linkedin and contact_info.linkedin.strip()}%
            \enspace $\cdot$ \enspace \faLinkedin \enspace \href{\VAR{contact_info.linkedin}}{\VAR{contact_info.linkedin_display or contact_info.linkedin_handle}}%
        \BLOCK{endif}%
    }
\end{center}
\vspace{0.4em}

% --- Sections ---
\BLOCK{for section in sections}
    \BLOCK{set section_type = section.get('type', '').lower()}
    \BLOCK{if not section_type}
        \BLOCK{if section.name|lower == "experience"}
            \BLOCK{set section_type = "experience"}
        \BLOCK{elif section.name|lower == "education"}
            \BLOCK{set section_type = "education"}
        \BLOCK{endif}
    \BLOCK{endif}

    \#{ Detect executive highlight sections }
    \BLOCK{set is_exec_summary = 'executive summary' in section.name|lower}
    \BLOCK{set is_key_achievements = 'key achievements' in section.name|lower}

    \section*{\VAR{section.name}}

    \BLOCK{if section_type == "text"}
        \BLOCK{if is_exec_summary}
            \begin{execbox}
                \VAR{section.content | markdown_links | markdown_formatting}
            \end{execbox}\vspace{0.5em}
        \BLOCK{else}
            \small{\VAR{section.content | markdown_links | markdown_formatting}}\vspace{0.5em}
        \BLOCK{endif}

    \BLOCK{elif section_type == "bulleted-list"}
        \BLOCK{if section.content and section.content|length > 0}
            \BLOCK{if is_key_achievements}
                \begin{execbox}
                    \begin{itemize}[leftmargin=1.5em,label=\textbullet,itemsep=0.1em,parsep=0pt]
                        \BLOCK{for item in section.content}
                            \rbullet{\VAR{item | markdown_links | markdown_formatting}}
                        \BLOCK{endfor}
                    \end{itemize}
                \end{execbox}\vspace{0.5em}
            \BLOCK{else}
                \begin{itemize}[leftmargin=1.5em,label=\textbullet,itemsep=0.1em,parsep=0pt]
                    \BLOCK{for item in section.content}
                        \rbullet{\VAR{item | markdown_links | markdown_formatting}}
                    \BLOCK{endfor}
                \end{itemize}\vspace{0.5em}
            \BLOCK{endif}
        \BLOCK{endif}

    \BLOCK{elif section_type == "inline-list"}
        \BLOCK{set formatted_items = []}
        \BLOCK{for item in section.content}
            \BLOCK{set _ = formatted_items.append(item | markdown_links | markdown_formatting)}
        \BLOCK{endfor}
        \rhorizontalbullets{\VAR{" \\textbullet\\ ".join(formatted_items)}}

    \BLOCK{elif section_type == "dynamic-column-list"}
        \BLOCK{set items = section.content}
        \BLOCK{set num_cols = 4}
        \BLOCK{set items_per_row = (items|length / num_cols)|round(0, 'ceil')|int}
        \small
        \begin{center}
        \begin{tabular*}{\linewidth}[t]{@{\extracolsep{\fill}}*{4}{p{0.22\linewidth}}@{}}
            \BLOCK{for row in range(items_per_row)}
                \BLOCK{set row_items = []}
                \BLOCK{for col in range(num_cols)}
                    \BLOCK{set item_index = row * num_cols + col}
                    \BLOCK{if item_index < items|length}
                        \BLOCK{set _ = row_items.append(items[item_index] | markdown_links | markdown_formatting)}
                    \BLOCK{else}
                        \BLOCK{set _ = row_items.append('')}
                    \BLOCK{endif}
                \BLOCK{endfor}
                \VAR{' & '.join(row_items)} \\
            \BLOCK{endfor}
        \end{tabular*}
        \end{center}
        \par\vspace{0.5em}

    \BLOCK{elif section_type == "icon-list"}
        \BLOCK{if section.content and section.content|length > 0}
            \begin{itemize}[leftmargin=1.5em,label=\textbullet,itemsep=0.1em,parsep=0pt]
                \BLOCK{for item in section.content}
                    \BLOCK{if item is mapping}
                        \rbullet{\VAR{item.certification | markdown_links | markdown_formatting} (\VAR{item.issuer | markdown_links | markdown_formatting}, \VAR{item.date | markdown_links | markdown_formatting})}
                    \BLOCK{else}
                        \rbullet{\VAR{item | markdown_links | markdown_formatting}}
                    \BLOCK{endif}
                \BLOCK{endfor}
            \end{itemize}\vspace{0.5em}
        \BLOCK{endif}

    \BLOCK{elif section_type == "grouped-list"}
        \BLOCK{for group in section.content}
            \BLOCK{if group is mapping and group.label is defined}
                \BLOCK{set group_items = group['items']}
                \textbf{\VAR{group.label | markdown_links | markdown_formatting}:} \VAR{group_items | join(', ') | markdown_links | markdown_formatting} \\[2pt]
            \BLOCK{else}
                \VAR{group | markdown_links | markdown_formatting} \\[2pt]
            \BLOCK{endif}
        \BLOCK{endfor}
        \vspace{0.5em}

    \BLOCK{elif section_type == "experience"}
        \BLOCK{for job in section.content}
            \needspace{4\baselineskip}
            \rjobtitle{\VAR{job.title | markdown_links | markdown_formatting}}
            \rcompanydate{\textit{\VAR{job.company | markdown_links | markdown_formatting}}}{\VAR{job.dates | markdown_links | markdown_formatting}}
            \BLOCK{if job.description and job.description|length > 0}
                \begin{itemize}[leftmargin=1.5em,label=\textbullet,itemsep=0.15em,parsep=0pt,topsep=0.1em]
                    \BLOCK{for bullet in job.description}
                        \rbullet{\VAR{bullet | markdown_links | markdown_formatting}}
                    \BLOCK{endfor}
                \end{itemize}
            \BLOCK{endif}
            \BLOCK{if not loop.last}\vspace{0.4em}\BLOCK{endif}
        \BLOCK{endfor}
        \vspace{0.5em}

    \BLOCK{elif section_type == "education"}
        \BLOCK{for edu in section.content}
            \needspace{3\baselineskip}
            \reducationentry{\VAR{edu.degree | markdown_links | markdown_formatting}}{\VAR{edu.school | markdown_links | markdown_formatting}}{\VAR{edu.year | markdown_links | markdown_formatting}}
            \BLOCK{if edu.gpa}
                \noindent\small{GPA: \VAR{edu.gpa | markdown_links | markdown_formatting}}\par
            \BLOCK{endif}
            \BLOCK{if edu.coursework}
                \noindent\small{\textbf{Relevant Coursework:} \VAR{edu.coursework | join(', ') | markdown_links | markdown_formatting}}\par
            \BLOCK{endif}
            \BLOCK{if not loop.last}\vspace{0.4em}\BLOCK{endif}
        \BLOCK{endfor}
        \vspace{0.5em}

    \BLOCK{else}
        \textit{Unsupported section type: \VAR{section_type}}\vspace{0.5em}
    \BLOCK{endif}
\BLOCK{endfor}


\end{document}
